1045 /**
1046 A class to measure time elapsed. Provided in CS314 class.
1047 */
1048
1049 public static class InternalStopwatch
1050 {
1051 private long startTime;
1052 private long stopTime;
1053
1054 public static final double NANOS_PER_SEC = 1000000000.0;
1055
1056 /**
1057 start the stop watch.
1058 */
1059 public void start(){
1060 startTime = System.nanoTime();
1061 }
1062
1063 /**
1064 stop the stop watch.
1065 */
1066 public void stop()
1067 { stopTime = System.nanoTime(); }
1068
1069 /**
1070 elapsed time in seconds.
1071 @return the time recorded on the stopwatch in seconds
1072 */
1073 public double time()
1074 { return (stopTime - startTime) / NANOS_PER_SEC; }
1075
1076 public String toString(){
1077 return "elapsed time: " + time() + " seconds.";
1078 }
1079
1080 /**
1081 elapsed time in nanoseconds.
1082 @return the time recorded on the stopwatch in nanoseconds
1083 */
1084 public long timeInNanoseconds()
1085 { return (stopTime - startTime); }
1086 }
